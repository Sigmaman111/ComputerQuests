Sign in / Sign up

Sign in with your email (type a Gmail if you prefer) or use Google single-click sign in. This is a static frontend using Firebase Authentication.

This demo uses Firebase Authentication (Email/Password + Google). Add your domain to Firebase Authorized domains.
// ==================================================================================== // Initialize Firebase firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); // UI elements const emailEl = document.getElementById('email'); const passEl = document.getElementById('password'); const signInBtn = document.getElementById('sign-in'); const signUpBtn = document.getElementById('sign-up'); const googleBtn = document.getElementById('google-signin'); const errorEl = document.getElementById('error'); const authForms = document.getElementById('auth-forms'); const userPanel = document.getElementById('user-panel'); const userName = document.getElementById('user-name'); const userEmail = document.getElementById('user-email'); const userUid = document.getElementById('user-uid'); const signOutBtn = document.getElementById('sign-out'); const deleteBtn = document.getElementById('delete-account'); // Helper to show errors function showError(message){ errorEl.textContent = message || ''; } // Sign up with email & password signUpBtn.addEventListener('click', async () => { showError(''); const email = (emailEl.value || '').trim(); const password = passEl.value || ''; if(!email || !password){ showError('Please enter email and password.'); return; } try { const cred = await auth.createUserWithEmailAndPassword(email, password); // user automatically signed in after sign up console.log('Signed up:', cred.user.uid); } catch (err) { showError(err.message); } }); // Sign in with email & password signInBtn.addEventListener('click', async () => { showError(''); const email = (emailEl.value || '').trim(); const password = passEl.value || ''; if(!email || !password){ showError('Please enter email and password.'); return; } try { await auth.signInWithEmailAndPassword(email, password); } catch (err) { showError(err.message); } }); // Sign in with Google (popup) googleBtn.addEventListener('click', async () => { showError(''); const provider = new firebase.auth.GoogleAuthProvider(); try { await auth.signInWithPopup(provider); } catch (err) { showError(err.message); } }); // Sign out signOutBtn.addEventListener('click', async () => { await auth.signOut(); }); // Delete account (requires recent login; may fail if session is old) deleteBtn.addEventListener('click', async () => { const user = auth.currentUser; if(!user){ showError('No signed-in user.'); return; } if(!confirm('Delete this account? This action cannot be undone.')) return; try { await user.delete(); } catch (err) { // If the user needs to reauthenticate, inform them showError(err.message || 'Unable to delete account. Try signing-in again and retry.'); } }); // Auth state listener auth.onAuthStateChanged(user => { if(user){ // show user panel authForms.classList.add('hidden'); userPanel.classList.remove('hidden'); userName.textContent = user.displayName || '(No display name)'; userEmail.textContent = user.email || ''; userUid.textContent = 'UID: ' + user.uid; } else { // show sign form authForms.classList.remove('hidden'); userPanel.classList.add('hidden'); userName.textContent = '—'; userEmail.textContent = '—'; userUid.textContent = ''; } showError(''); }); // Optional: prefill email with a gmail suggestion if desired (function prefillGmailHint(){ // This is a convenience only; remove if not wanted. if(!emailEl.value) emailEl.value = ''; })();
